create table staff(
staff_id serial primary key,
first_name varchar(40) not null,
last_name varchar(40) not null,
birth_date Date check(birth_date > '1990-01-01'),
joined_date Date check(joined_date > birth_date),
salary numeric check(salary > 0)

)

--create check constraint on existing table

create table prices(
	price_id serial primary key,
	product_id int not null,
	price numeric not null,
	discount numeric not null,
	valid_from date not null

)

--create check constraint.......
-- price > 0, discount >=0 and price > discount

alter table prices add constraint prices_check
check(
price > 0
and discount >=0
and price > discount
)

--add data

--rename the existing check constraint..................

--alter table table_name rename constarint old_name to newname
